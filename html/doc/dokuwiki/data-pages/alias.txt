
====== Alias ======

  $ alias blabber=echo # define an alias
  $ unalias blabber # remove an alias
  $ alias # list all command aliases
  alias egrep='egrep --color=auto'
  alias fgrep='fgrep --color=auto'
  alias grep='grep --color=auto'
  alias l='ls -CF'
  alias la='ls -A'
  alias ll='ls -alF'
  alias ls='ls --color=auto'
